"3.2 ()匹配{{{
inoremap ( ()i

function SkipParen()
	let temp=getline('.')[col('.')-1]
	if temp==")"
		return "\la"
	else
		return ")"
	endif
endfunction

inoremap ) =SkipParen()
"}}}
"3.3 ""匹配{{{
function SkipQuotation()
	let temp=getline('.')[col('.')-1]
	if temp=='"'
		return "\la"
	else
		return '""'."\i"
	endif
endfunction

inoremap " =SkipQuotation()
"}}}
"3.4 ''匹配{{{
function SkipOtherQuotation()
	let temp=getline('.')[col('.')-1]
	if temp=="'"
		return "\la"
	else
		return "''\i"
	endif
endfunction

inoremap ' =SkipOtherQuotation()
"}}}
"3.5 <>匹配{{{
inoremap < <>i

function SkipBook()
	let temp=getline('.')[col('.')-1]
	if temp==">"
		return "\la"
	else
		return ">"
	endif
endfunction

inoremap > =SkipBook()
"}}}

"5.1 Java 注释{{{
function JavaAnnotation()
let temp=getline(".")
let aTemp=temp[0].temp[1]
if aTemp ==# "//"
normal ^xx
else
normal I//
endif
endfunction

:autocmd FileType java nnoremap a :call JavaAnnotation()
"}}}
"5.2 Java 编译设置{{{
set makeprg=javac\ @%
:autocmd FileType java nnoremap  :!javac %
"}}}
"5.3 运行Java{{{

"}}}
